/**
 * Strength of intact rock material rating.
 * @param {String} idx selected index either 'pls' for point-loads strength or 'ucs' for uniaxial compressive strength
 * @param {Number} strength strength of intact rock material (in MPa)
 * @returns {Number} 
 */
export function CalcR1(idx, strength){
    let val_r1 = 0;
    if(idx == "pls"){
        if(strength >= 10){
            val_r1 = 15
        } else if(10 > strength >= 4){
            val_r1 = 12
        } else if(4 > strength >= 2){
            val_r1 = 7
        } else if(2 > strength >= 1){
            val_r1 = 4
        } else {
            val_r1 = "For value lower than 1 MPa, please proceed with Uniaxial Compressive Strength Test"
        }
    } else if(idx == "ucs"){
        if(strength >= 250){
            val_r1 = 15
        } else if(250 > strength >= 100){
            val_r1 = 12
        } else if(100 > strength >= 50){
            val_r1 = 7
        } else if(50 > strength >= 25){
            val_r1 = 4
        } else if(25 > strength >= 5){
            val_r1 = 2
        } else if(5 > strength >= 1){
            val_r1 = 1
        } else if(strength < 1){
            val_r1 = 0
        }
    }
    return val_r1
}


/**
 * Drill core RQD rating.
 * @param {Number} drillcoreRQD drill core quality or rock quality designation (in percent)
 * @returns {Number}
 */
export function CalcR2(drillcoreRQD){
    let val_r2 = 0;
    if(100 >= drillcoreRQD >= 90){
        val_r2 = 20
    } else if(90 > drillcoreRQD >= 75){
        val_r2 = 17
    } else if(75 > drillcoreRQD >= 50){
        val_r2 = 13
    } else if(50 > drillcoreRQD >= 25){
        val_r2 = 8
    } else if(drillcoreRQD < 25){
        val_r2 = 3
    }
    return val_r2 
}


/**
 * Space of discontinuity rating.
 * @param {Number} spacing value of rock spacing (in m, float/decimal)
 * @returns {Number}
 */
export function CalcR3(spacing){
    let val_r3 = 0;
    if(spacing >= 2.0){
        val_r3 = 20
    } else if(2.0 > spacing >= 0.6){
        val_r3 = 15
    } else if(0.6 > spacing >= 0.2){
        val_r3 = 10
    } else if(0.2 > spacing >= 0.06){
        val_r3 = 8
    } else if(spacing < 0.06){
        val_r3 = 5
    }
    return val_r3
}
